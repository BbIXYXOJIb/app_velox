53class HumanHasher(object): 

54 

55 """ 

56 Transforms hex digests to human-readable strings. 

57 

58 The format of these strings will look something like: 

59 `victor-bacon-zulu-lima`. The output is obtained by compressing the input 

60 digest to a fixed number of bytes, then mapping those bytes to one of 256 

61 words. A default wordlist is provided, but you can override this if you 

62 prefer. 

63 

64 As long as you use the same wordlist, the output will be consistent (i.e. 

65 the same digest will always render the same representation). 

66 """ 

67 

68 def __init__(self, wordlist=DEFAULT_WORDLIST): 

69 if len(wordlist) != 256: 

70 raise ArgumentError("Wordlist must have exactly 256 items") 

71 self.wordlist = wordlist 

72 

73 def humanize(self, hexdigest, words=4, separator='-'): 

74 

75 """ 

76 Humanize a given hexadecimal digest. 

77 

78 Change the number of words output by specifying `words`. Change the 

79 word separator with `separator`. 

80 

81 >>> digest = '60ad8d0d871b6095808297' 

82 >>> HumanHasher().humanize(digest) 

83 'sodium-magnesium-nineteen-hydrogen' 

84 """ 

85 

86 # Gets a list of byte values between 0-255. 

87 bytes = [int(x, 16) for x in list(map(''.join, list(zip(hexdigest[::2], hexdigest[1::2]))))] 

88 # Compress an arbitrary number of bytes to `words`. 

89 compressed = self.compress(bytes, words) 

90 # Map the compressed byte values through the word list. 

91 return separator.join(self.wordlist[byte] for byte in compressed) 

92 

93 @staticmethod 

94 def compress(bytes, target): 

95 

96 """ 

97 Compress a list of byte values to a fixed target length. 

98 

99 >>> bytes = [96, 173, 141, 13, 135, 27, 96, 149, 128, 130, 151] 

100 >>> HumanHasher.compress(bytes, 4) 

101 [205, 128, 156, 96] 

102 

103 Attempting to compress a smaller number of bytes to a larger number is 

104 an error: 

105 

106 >>> HumanHasher.compress(bytes, 15) # doctest: +ELLIPSIS 

107 Traceback (most recent call last): 

108 ... 

109 ValueError: Fewer input bytes than requested output 

110 """ 

111 

112 length = len(bytes) 

113 if target > length: 

114 raise ValueError("Fewer input bytes than requested output") 

115 

116 # Split `bytes` into `target` segments. 

117 seg_size = length // target 

118 segments = [bytes[i * seg_size:(i + 1) * seg_size] 

119 for i in range(target)] 

120 # Catch any left-over bytes in the last segment. 

121 segments[-1].extend(bytes[target * seg_size:]) 

122 

123 # Use a simple XOR checksum-like function for compression. 

124 checksum = lambda bytes: reduce(operator.xor, bytes, 0) 

125 checksums = list(map(checksum, segments)) 

126 return checksums

128 def uuid(self, **params): 

129 

130 """ 

131 Generate a UUID with a human-readable representation. 

132 

133 Returns `(human_repr, full_digest)`. Accepts the same keyword arguments 

134 as :meth:`humanize` (they'll be passed straight through). 

135 """ 

136 

137 digest = str(uuidlib.uuid4()).replace('-', '') 

138 return self.humanize(digest, **params), digest
